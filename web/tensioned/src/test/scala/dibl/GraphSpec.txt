package dibl

import org.scalatest._

class GraphSpec extends FlatSpec with Matchers {
  "4x4-0" should "produce nodes and links" in {
    val data = Graph.get()
    data.get("nodes").size should equal (16*16)
    data.get("links").size should equal (7*2*3*6)
  }
}
